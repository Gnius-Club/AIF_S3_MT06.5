# --- Configuraci√≥n del bus I2C y el sensor BME280 ---
# I2C(ID, SCL, SDA, FREQ) -- Pines de I2C en Micro:bit son P19 (SCL) y P20 (SDA)

# Definir pines para LEDs y Buzzer
buzzer = Pin(Pin.P0, Pin.OUT) # Buzzer en P0
led_azul = Pin(Pin.P1, Pin.OUT) # LED azul en P1 (indicador OK)
led_rojo = Pin(Pin.P8, Pin.OUT) # LED rojo en P8 (indicador ALERTA)

# --- Definici√≥n de Umbrales de Seguridad ---

TEMP_MAX = 28.0

HUM_MAX = 80.0

# --- Funci√≥n de Configuraci√≥n Inicial ---
def setup_sistema():
    display.scroll("Alerta Invernadero ON")
    buzzer.write_digital(0) # Asegurar que el buzzer est√© apagado al inicio
    led_azul.write_digital(0) # Apagar LEDs al inicio
    led_rojo.write_digital(0)
    time.sleep(1)
    display.clear()

# Ejecutar configuraci√≥n
setup_sistema()

# --- Bucle principal de monitoreo ---
while True:
    # Limpiar datos anteriores de display
    display.clear()
    led_azul.write_digital(0)
    led_rojo.write_digital(0)
    buzzer.write_digital(0)
    # Leer los datos del sensor BME280
    try:
        temperatura = float(bme.read_temperature().strip('C'))
        humedad = float(bme.read_humidity().strip('%'))
        presion = float(bme.read_pressure().strip('hPa')) # No se usa directamente para alerta, pero se lee
    except ValueError as e:
        print("Error al leer/convertir sensor:", e)
        display.show(Image.SAD) # Mostrar error visual
        time.sleep(2)
        continue # Saltar esta iteraci√≥n y probar de nuevo
    # Imprimir lecturas para depuraci√≥n en la consola serial (opcional, muy √∫til)
    print(f"Temp: {temperatura}¬∞C, Hum: {humedad}%, Pres: {presion}hPa")
    # --- L√≥gica de Alerta de Umbrales ---
    # Condici√≥n de ALERTA: Si la temperatura o la humedad est√°n FUERA de su rango seguro
    if (temperatura < TEMP_MIN or temperatura > TEMP_MAX or \
        humedad < HUM_MIN or humedad > HUM_MAX):

        display.show(Image.ANGRY) # Icono de alerta
        print("¬°ALERTA AMBIENTAL! Valores fuera de rango.")

        # Activar alarma local (parpadeo r√°pido de LED rojo y sonido intermitente del buzzer)
        for _ in range(5): # 5 parpadeos/sonidos
            led_rojo.write_digital(1)
            buzzer.write_digital(1)
            time.sleep_ms(100)
            led_rojo.write_digital(0)
            buzzer.write_digital(0)
            time.sleep_ms(100)

    # Condici√≥n de NORMALIDAD: Si AMBAS temperatura y humedad est√°n DENTRO de su rango seguro
    elif (temperatura >= TEMP_MIN and temperatura <= TEMP_MAX and \
          humedad >= HUM_MIN and humedad <= HUM_MAX):

        display.show(Image.HAPPY) # Icono de OK
        led_azul.write_digital(1) # Encender LED azul fijo
        print("Condiciones ambientales OK.")
    else: # Este else cubrir√≠a casos donde no se cumple ninguna de las anteriores (ej. rangos solapados, errores de lectura)
        display.show(Image.QUESTION)
        print("Estado ambiental desconocido o no categorizado.")

    time.sleep(2) # Esperar 2 segundos antes de la siguiente lectura/evaluaci√≥n
